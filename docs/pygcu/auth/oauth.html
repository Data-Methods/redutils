162    def refresh_token(self, sleep: bool = False) -> Optional[str]:
163        """Refreshes token based on set value for authorized_url and provided credentials
164
165        :param sleep: (bool, defaults=False) - enforce a thread-stopping wait before refreshing
166
167        :return: str
168
169        """
170        if sleep:
171            time.sleep(60)
172        headers = {"Content-Type": "application/x-www-form-urlencoded"}
173        if self._api_internal_timer is None:
174            self._api_internal_timer = datetime.now()
175
176        try:
177            resp = self._auth_session.post(
178                str(self.authorize_url), headers=headers, data=self._auth_data
179            )
180            if 200 > resp.status_code > 299:
181                Exit(LEVEL_ERROR, f"Unable to refresh token: {resp.status_code}")
182        except Exception as e:
183            Exit(LEVEL_CRITICAL, f"{e}\n{traceback.format_exc()}")
184
185        try:
186            payload = resp.json()
187            token = payload.get("access_token")
188            expires_in = payload.get("expires_in")
189        except Exception as e:
190            Exit(LEVEL_CRITICAL, f"{traceback.format_exc()}")
191
192        if not token or not expires_in:
193            Exit(LEVEL_ERROR, "Refresh token failed")
194
195        self._api_token = token
196        self._api_expires_in = expires_in
197        self._api_internal_timer = datetime.now()
198        Red.log("Token refreshed")
199        return self._api_token
